创建型模式 - 工厂

# 如何理解工厂模式

  • 根据传入参数不同，可以创建不同的对象
  • 返回一个新对象

# 工厂模式的实现

# ES5方式

// 简单工厂模式
function Car(name, price) {
    this.name = name;
    this.price = price;
}
Car.prototype.show = function() {
    console.log(`汽车品牌是: ${this.name}, 价格: ${this.name}`)
};

function Factory(type) {
    switch(type) {
        case 'a':
            return new Car('a', 199999);
            break;
        case 'b':
            return new Car('b', 233333333);
            break;
    }
}
new Factory('a');

// 工厂方式模式
function Factory(type) {
    // 不论是否使用new, 创建的都是实例
    if (this instanceof Factory) {
        return this[type]();
    }
    return new Factory(type);
}
Factory.prototype.a = function () {
    this.name = 'a';
    this.price = '199999';
}
Factory.prototype.b = function () {
    this.name = 'b';
    this.price = '233333333';
}
Factory.prototype.c = function () {
    this.name = 'c';
    this.price = '1000000';
}
Factory.prototype.show = function () {
    console.log(`汽车品牌是: ${this.name}, 价格: ${this.name}`)
}

let a = Factory('a');
a.show();

let b = new Factory('b');
b.show();

# ES6方式

class User {
  constructor(name = '', viewPage = []) {
    if(new.target === User) { // new.target
      throw new Error('抽象类不能实例化!');
    }
    this.name = name;
    this.viewPage = viewPage;
  }
}

class UserFactory extends User {
  constructor(name, viewPage) {
    super(name, viewPage)
  }
  create(role) {
    switch (role) {
      case 'superAdmin':
        return new UserFactory( '超级管理员', ['首页', '通讯录', '发现页', '应用数据', '权限管理'] );
        break;
      case 'admin':
        return new UserFactory( '普通用户', ['首页', '通讯录', '发现页'] );
        break;
      case 'user':
        return new UserFactory( '普通用户', ['首页', '通讯录', '发现页'] );
        break;
      default:
        throw new Error('参数错误, 可选参数:superAdmin、admin、user')
    }
  }
}

let userFactory = new UserFactory();
let superAdmin = userFactory.create('superAdmin');
let admin = userFactory.create('admin');
let user = userFactory.create('user');
